from graphics import *

def main():
    win = GraphWin("Runner", 1000, 500, autoflush=False)
    win.setBackground('blue')
    #verifica teclas
    key = win.checkKey()

    carregarArquivos=1
    playerVivo=1
    cenarioCriado=0
    cachorroCriado=0
    carroPode=0
    skateCriado=0
    playerCriado=0
    layoutCriado=0
    lixoCriado=0


    while key != "Escape":
        #SISTEMA DE RESTART
        if (playerVivo == 0 and key=="return") or key=="r" or carregarArquivos==1:
            # JOGO
            jogoPausado = 0
            jogoPausado_cont = 0
            podeJogar = 1

            # CARRO
            if carroPode==1:
                carro.undraw()
            carro = Image(Point(1500, 587), "carro.png")
            carro.draw(win)
            carroPode = 1
            carro_cont = 0

            # CENARIO
            if cenarioCriado==1:
                cenario.undraw()
                paralax.undraw()
            paralax = Image(Point(1000, 250), "paralax.png")
            paralax.draw(win)
            cenario = Image(Point(15000, 250), "cenario.png")
            cenario.draw(win)
            cenarioMovimento = 1
            cenarioLocal = 0
            cenarioCriado = 1

            # ANIMAÇÃO CACHORRO
            if cachorroCriado==1:
                imagem.undraw()
            cachorro = 0            #contador do cachorro
            imagem = Image(Point(400 + 3000, 390), "cachorro1.png")
            centro = imagem.getAnchor()
            localCachorro = 0
            imagem.draw(win)
            cachorroPosEsq = 350 + 3000
            cachorroPosDir = 450 + 3000
            cachorroCriado=1

            # SKATE
            if skateCriado==1:
                skate.undraw()
            skate = Image(Point(500 + 1000, 370), "skate.png")
            skateCentro = skate.getAnchor()
            skate.draw(win)
            skate_cont = 0
            skateMove = 1
            skate_pos = 465 + 1000
            skate_posDir = 550 + 1000
            desenharSkatistaMorto = 1
            skateCriado = 1

            # PLAYER
            if playerCriado==1:
                player.undraw()
            podeTrocarVida = 2
            playerVida = 3
            playerVivo = 1
            delay_para_pular = 0
            limitadorDeAltura_cont = 0
            limitadorDeAltura = 0
            altura = 0
            player_cont = 0
            player = Image(Point(200, 390), "sprite1.png")
            playerCentro = player.getAnchor()
            player.draw(win)
            posicao = 0
            posicaoPlayer = 264
            posicaoPlayerEsq = 133
            playerCriado=1

            # layout
            if layoutCriado==1:
                layout.undraw()
            layout = Image(Point(500, 250), "layout3.png")
            layout.draw(win)
            layoutCriado=1

            # LIXO
            if lixoCriado==1:
                lixo.undraw()
            lixo = Image(Point(100+600, 385), "lixo.png")
            lixo.draw(win)
            lixo_cont = 0
            lixoMove = 1
            lixoPosicaoEsq = 96+600
            lixoPosicaoDir = 123+600
            lixoCriado=1

            #Para evitar bug
            carregarArquivos=0




        key = win.checkKey()
        if playerVivo==1:
            update(120)
        else:
            update(30)

        #SISTEMA DE COLISÃO
            #COLISAO ENTRE PLAYER E LIXO
        if ((posicaoPlayerEsq<=lixoPosicaoEsq and lixoPosicaoEsq<=posicaoPlayer) or (posicaoPlayerEsq<=lixoPosicaoDir and lixoPosicaoDir<=posicaoPlayer))  and posicao==1 and altura<10:
            playerVida=playerVida-1
            lixoMove = 0
            lixoPosicaoDir=100000
            lixoPosicaoEsq=100000


            #COLISAO ENTRE PLAYER E CACHORRO
        if ((posicaoPlayerEsq<=cachorroPosEsq and cachorroPosEsq<=posicaoPlayer) or (posicaoPlayerEsq<=cachorroPosDir and cachorroPosDir<=posicaoPlayer))  and posicao==0 and altura<10:
            playerVida=playerVida-1
            lixoMove = 0
            cachorroPosDir=100000
            cachorroPosEsq=100000

        #SISTEMA DE COLISÃO ENTRE JOGADOR E SKATISTA
        if ((posicaoPlayerEsq<=skate_pos and skate_pos<=posicaoPlayer) or (posicaoPlayerEsq<=skate_posDir and skate_posDir<=posicaoPlayer)) and posicao==0:
            playerVida=playerVida-1
            skateMove=0
            skate_pos=100000
            skate_posDir=100000

        #SISTEMA DE MORTE
        if playerVida==0:
            if podeTrocarVida==0:
                #LAYOUT MORTO
                layout.undraw()
                layout = Image(Point(500, 250), "morto.png")
                layout.draw(win)
                #PLAYER MORTO
                player.undraw()
                playerCentro = player.getAnchor()
                player = Image(playerCentro, "playerMorto.png")
                player.draw(win)
                podeTrocarVida = 3

            playerVivo=0
            posicaoPlayer=0
            posicaoPlayerEsq=0
            skate_cont+=1

        #LAYOUT
        if playerVida==2 and podeTrocarVida==2:
            layout.undraw()
            layout = Image(Point(500, 250), "layout2.png")
            layout.draw(win)
            podeTrocarVida=1

        if playerVida==1 and podeTrocarVida==1:
            layout.undraw()
            layout = Image(Point(500, 250), "layout1.png")
            layout.draw(win)
            podeTrocarVida=0

        #SISTEMA DE PAUSE
        if (jogoPausado == 1 or jogoPausado==0) and jogoPausado_cont<20:
            jogoPausado_cont += 1

        if key == "p" or key=="P":
            if jogoPausado == 1:
                if jogoPausado_cont>10:
                    podeJogar = 1
                    jogoPausado=0
                    jogoPausado_cont = 0

            if jogoPausado == 0:
                if jogoPausado_cont>10:
                    podeJogar=0
                    jogoPausado=1
                    jogoPausado_cont=0




        #VARIAVEL PARA PODER PAUSAR
        if podeJogar==1:
            #CÓDIGOS ABAIXO PARA FUNCIONAR A VARIÁVEL DE PAUSE
            # lixo
            if lixoMove == 1 and playerVivo == 1:
                lixo.move(-13, 0)
                lixo_cont += 1
                lixoPosicaoDir = lixoPosicaoDir - 13
                lixoPosicaoEsq = lixoPosicaoEsq - 13
                if lixo_cont > 150:
                    lixo.undraw()
                    lixo = Image(Point(100+1000, 385), "lixo.png")
                    lixo.draw(win)
                    lixo_cont = 0
                    lixoPosicaoEsq = 96+1000
                    lixoPosicaoDir = 123+1000

            if lixoMove == 0:
                lixo.undraw()
                if playerVivo == 1:
                    lixo.move(-13, 0)
                lixoCentro = lixo.getAnchor()
                lixo = Image(lixoCentro, "lixoCaido.png")
                lixo.draw(win)
                lixo_cont += 1

                if lixo_cont > 150:
                    lixoMove = 1

            #ANIMAÇÃO DO JOGADOR----------

            #SISTEMA DE PULO
            if delay_para_pular>0:
                limitadorDeAltura = 1

            if limitadorDeAltura == 1:
                delay_para_pular+=1

            if delay_para_pular>30:
                limitadorDeAltura=0
                delay_para_pular=0

            if altura>0:
                player.move(0, 10)
                playerCentro = player.getAnchor()
                player.undraw()
                player = Image(playerCentro , "pulando.png")
                player.draw(win)
                altura -= 10

            elif limitadorDeAltura==0 and key == "space" and playerVivo==1:
                player.move(0, -150)
                playerCentro = player.getAnchor()
                player.undraw()
                player = Image(playerCentro , "pulando.png")
                player.draw(win)
                altura += 150
                delay_para_pular += 1

            #VAI PRA DIREITA
            elif (key == "Right" or key == "d" or key == "D") and posicaoPlayer<1000:
                if playerVivo==1:
                    if player_cont < 3:
                        player.move(4, 0)
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro , "sprite1.png")
                        player.draw(win)
                        posicaoPlayerEsq = posicaoPlayerEsq + 4
                        posicaoPlayer = posicaoPlayer + 4
                        player_cont+=1


                    if player_cont >= 3 and player_cont < 6:
                        player.move(4, 0)
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro , "sprite2.png")
                        player.draw(win)
                        posicaoPlayerEsq = posicaoPlayerEsq + 4
                        posicaoPlayer = posicaoPlayer + 4
                        player_cont += 1


                    if player_cont >= 6 and player_cont < 9:
                        player.move(4, 0)
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro , "sprite3.png")
                        player.draw(win)
                        posicaoPlayerEsq = posicaoPlayerEsq + 4
                        posicaoPlayer = posicaoPlayer + 4
                        player_cont += 1


                    if player_cont >= 9 and player_cont < 12:
                        player.move(4, 0)
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro , "sprite4.png")
                        player.draw(win)
                        posicaoPlayerEsq = posicaoPlayerEsq + 4
                        posicaoPlayer = posicaoPlayer + 4
                        player_cont += 1

                    if player_cont == 12:
                        player_cont = 0

            elif (key == "Left" or key == "a" or key == "A") and posicaoPlayerEsq>0:
                if playerVivo == 1:
                    if player_cont < 3:
                        player.move(-4, 0)
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro, "sprite1.png")
                        player.draw(win)
                        posicaoPlayer = posicaoPlayer - 4
                        posicaoPlayerEsq = posicaoPlayerEsq - 4
                        player_cont += 1

                    if player_cont >= 3 and player_cont < 6:
                        player.move(-4, 0)
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro, "sprite2.png")
                        player.draw(win)
                        posicaoPlayer = posicaoPlayer - 4
                        posicaoPlayerEsq = posicaoPlayerEsq - 4
                        player_cont += 1


                    if player_cont >= 6 and player_cont < 9:
                        player.move(-4, 0)
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro, "sprite3.png")
                        player.draw(win)
                        posicaoPlayer = posicaoPlayer - 4
                        posicaoPlayerEsq = posicaoPlayerEsq - 4
                        player_cont += 1

                    if player_cont >= 9 and player_cont < 12:
                        player.move(-4, 0)
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro, "sprite4.png")
                        player.draw(win)
                        posicaoPlayer = posicaoPlayer - 4
                        posicaoPlayerEsq = posicaoPlayerEsq - 4
                        player_cont += 1

                    if player_cont == 12:
                        player_cont = 0

            elif key == "w" or key == "W" or key == "Up":
                if posicao==0 and playerVivo==1:
                    player.move(-15,-30)
                    posicao+=1
                    posicaoPlayer=posicaoPlayer-15
                    posicaoPlayerEsq=posicaoPlayerEsq-15

            elif key == "s" or key == "S" or key == "Down":
                if posicao==1 and playerVivo ==1:
                    player.move(15,30)
                    posicao-=1
                    posicaoPlayer=posicaoPlayer+15
                    posicaoPlayerEsq=posicaoPlayerEsq+15

            else:
                if playerVivo==1:
                    if player_cont < 3:
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro , "sprite1.png")
                        player.draw(win)
                        player_cont+=1

                    if player_cont >= 3 and player_cont < 6:
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro , "sprite2.png")
                        player.draw(win)
                        player_cont += 1

                    if player_cont >= 6 and player_cont < 9:
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro , "sprite3.png")
                        player.draw(win)
                        player_cont += 1

                    if player_cont >= 9 and player_cont < 12:
                        playerCentro = player.getAnchor()
                        player.undraw()
                        player = Image(playerCentro , "sprite4.png")
                        player.draw(win)
                        player_cont += 1

                    if player_cont == 12:
                        player_cont = 0


            #cenario
            if cenarioMovimento==1 and playerVivo==1:
                cenario.move(-13,0)
                paralax.move(-0.2,0)
                cenarioLocal=cenarioLocal+13
                if cenarioLocal>29500:
                    podeJogar=0
                    cenarioMovimento=0

            #skate
            if skateMove==1:
                if skate_cont>=0:
                    skate.undraw()
                    skate.move(-20,0)
                    skateCentro = skate.getAnchor()
                    skate = Image(skateCentro, "skate.png")
                    skate.draw(win)
                    skate_pos=skate_pos-20
                    skate_posDir = skate_posDir-20
                    skate_cont += 1

                if skate_cont>300:
                    skate.undraw()
                    skate = Image(Point(500+1000, 370), "skate.png")
                    skate.draw(win)
                    skate_cont=0
                    skate_pos = 465 + 1000
                    skate_posDir = 550 + 1000

            if skateMove==0:
                if skate_cont>=0:
                    skate.undraw()
                    if playerVivo==1:
                        skate.move(-13,0)
                    if playerVivo==0:
                        if desenharSkatistaMorto==1:
                            skate2Centro=skate.getAnchor()
                            skate2 = Image(skate2Centro, "skatecaido.png")
                            skate2.draw(win)
                            desenharSkatistaMorto=0
                    skateCentro = skate.getAnchor()
                    skate = Image(skateCentro, "skatecaido.png")
                    skate.draw(win)
                    skate_cont += 1
                if skate_cont>500:
                    skateMove=1




            #ANIMAÇÃO DO CACHORRO
            #se o cachorro tiver longe, spawna ele de novo
            if localCachorro >500:
                imagem.undraw()
                imagem = Image(Point(400+1000, 380), "cachorro1.png")
                centro = imagem.getAnchor()
                localCachorro=0
                cachorro = 0
                cachorroPosEsq = 350 + 1000
                cachorroPosDir = 450 + 1000

            else:
                if cachorro<3:
                    imagem.move(-20,0)
                    centro = imagem.getAnchor()
                    imagem.undraw()
                    imagem = Image(centro, "cachorro1.png")
                    imagem.draw(win)
                    cachorro += 1
                    localCachorro+=1
                    cachorroPosDir=cachorroPosDir-20
                    cachorroPosEsq=cachorroPosEsq-20


                if cachorro>=3 and cachorro <6:
                    imagem.move(-20, 0)
                    centro = imagem.getAnchor()
                    imagem.undraw()
                    imagem = Image(centro, "cachorro2.png")
                    imagem.draw(win)
                    cachorro += 1
                    localCachorro += 1
                    cachorroPosDir=cachorroPosDir-20
                    cachorroPosEsq=cachorroPosEsq-20

                if cachorro>=6 and cachorro <9:
                    imagem.move(-20, 0)
                    centro = imagem.getAnchor()
                    imagem.undraw()
                    imagem = Image(centro, "cachorro3.png")
                    imagem.draw(win)
                    cachorro += 1
                    localCachorro += 1
                    cachorroPosDir=cachorroPosDir-20
                    cachorroPosEsq=cachorroPosEsq-20

                if cachorro==9:
                    cachorro=0


            #Animação do carro
            if carroPode==1:
                carro.move(-20,0)
                carro_cont += 1
                if carro_cont>400:
                    carro.undraw()
                    carro = Image(Point(1500, 587), "carro.png")
                    carro.draw(win)
                    carro_cont=0





    win.close()

main()
